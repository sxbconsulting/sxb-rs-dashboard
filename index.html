Weekly entries CREATE TABLE IF NOT EXISTS weekly_entries ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL, week_platform text NOT NULL, entry_data jsonb NOT NULL, updated_at timestamptz DEFAULT now(), created_at timestamptz DEFAULT now(), UNIQUE(user_id, week_platform) ); ALTER TABLE weekly_entries ENABLE ROW LEVEL SECURITY; CREATE POLICY "Users own entries" ON weekly_entries FOR ALL USING (auth.uid() = user_id); -- 2. Watch accounts CREATE TABLE IF NOT EXISTS watch_accounts ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL, account_data jsonb NOT NULL, created_at timestamptz DEFAULT now() ); ALTER TABLE watch_accounts ENABLE ROW LEVEL SECURITY; CREATE POLICY "Users own accounts" ON watch_accounts FOR ALL USING (auth.uid() = user_id); -- 3. Veille notes CREATE TABLE IF NOT EXISTS veille_notes ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL UNIQUE, notes text DEFAULT '', updated_at timestamptz DEFAULT now() ); ALTER TABLE veille_notes ENABLE ROW LEVEL SECURITY; CREATE POLICY "Users own notes" ON veille_notes FOR ALL USING (auth.uid() = user_id); -- 4. Competitors CREATE TABLE IF NOT EXISTS competitors ( id uuid DEFAULT gen_random_uuid() PRIMARY KEY, user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL, comp_data jsonb NOT NULL, created_at timestamptz DEFAULT now() ); ALTER TABLE competitors ENABLE ROW LEVEL SECURITY; CREATE POLICY "Users own competitors" ON competitors FOR ALL USING (auth.uid() = user_id); -- ✅ Tables créées avec Row Level Security activé -- Chaque utilisateur voit uniquement ses propres données
